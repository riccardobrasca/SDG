\chapter{The synthetic theory}

Let $R$ be a commutative ring.

\begin{definition}
  \label{SDG.D}
  \leanok
  \lean{SDG.D}
We denote by $D$ the set
\[
\{r \in R \mbox{ such that } r ^ 2 = 0 \}
\]
\end{definition}

\begin{lemma}
  \label{SDG.zero_mem_D}
  \leanok
  \lean{SDG.zero_mem_D}
  \uses{SDG.D}
  We have that $0 \in D$
\end{lemma}
\begin{proof}
  \leanok
  Obvious.
\end{proof}
